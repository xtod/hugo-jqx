{{- $url := .Destination -}} {{- $isRemote := strings.HasPrefix $url "http" -}} {{- $isAnchor := strings.HasPrefix $url "#" -}} {{- $isMail := strings.HasPrefix $url "mailto:" -}} {{- $isTel := strings.HasPrefix $url "tel:" -}} {{- $title := .Title -}} {{- $text := .Text -}} {{- $attrs := dict "href" $url -}} {{- if $title -}} {{- $attrs = merge $attrs (dict "title" $title) -}} {{- end -}} {{- if or $isRemote (and (not $isAnchor) (not $isMail) (not $isTel)) -}} {{- $attrs = merge $attrs (dict "target" "_blank" "rel" "noopener noreferrer") -}} {{- end -}} {{- if $isMail -}} {{- $attrs = merge $attrs (dict "rel" "noopener noreferrer") -}} {{- end -}} {{- $text | safeHTML -}} {{- if $isRemote -}} ↗ {{- end -}}
